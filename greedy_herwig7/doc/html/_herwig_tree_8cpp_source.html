595 bool HerwigTree::HardProc()
596 {
597  /* We need this to extract the pthat */
598  float minPt = 100000;
599  /* Counters and a storage for ttbar events */
600  int bCount = 0, qCount = 0;
601  vector<tPPtr> topLepts;
602 
603  /* The hardest subprocess
604  Hard process (method 1 - brings also the id 82 collimations)
605  const ParticleVector hardParts = event->primarySubProcess()->outgoing(); (const_iterator)
606  Hard process (method 2 - preferred, and used in this code) */
607  tPVector hardParts = mEvent->primaryCollision()->step(0)->getFinalState();
608 
609  for (auto &hard : hardParts) {
610  int absId = abs(hard->id());
611 
612  if (absId==82) continue;
613  auto part = OptimalParton(hard);
614 
615  #ifdef DEBUG
616  cerr << "id: " << absId << ": " << hard->number() << ": " << hard->momentum().perp()/GeV << endl;
617  cerr << absId << ": " << hard->momentum().perp()/GeV << " " << first.perp()/GeV << " " << second.perp()/GeV << endl;
618  #endif
619  minPt = min(minPt,part->momentum().perp()/GeV);
620 
621  if (mMode==2 and absId==22) {
622  if (!GammaAdd(part)) return false;
623  } else if (mMode==3 and absId==13) {
624  if (!MuonAdd(part)) return false;
625  } else if (mMode>=4 and absId==6) {
626  auto topc = Probe(part);
627  int wInd = -1;
628  for (auto &child : topc) {
629  auto childId = abs(child->id());
630  if (childId > 10 and childId < 20) {
631  topLepts.push_back(child);
632  } else if (childId==24) {
633  PartonAppend(PseudoJettify(child),child,(char) 2,-1);
634  wInd = child->number();
635  if (part->children().size()>0 and IsLepton(part->children()[0]->id()))
636  mLeptonFriend = part->number();
637  } else if (childId==5) {
638  ++bCount;
639  PartonAppend(PseudoJettify(child),child,(char) 0,-1);
640  PartonDescend(child);
641  BNeutrinos(child);
642  } else if (childId<5) {
643  ++qCount;
644  if (wInd < 0)
645  wInd = part->number();
646  PartonAppend(PseudoJettify(child),child,(char) 0,wInd);
647  PartonDescend(child);
648  } else if (childId==22) {
649  throw runtime_error("Illegal top photon activity!");
650  } else if (childId==21) { // This is odd
651  PartonAppend(PseudoJettify(child),child,(char) 0,-1);
652  PartonDescend(child);
653  } else {
654  throw runtime_error(Form("Unexpected t consequences! %ld",childId));
655  }
656  }
657 
658  if (wInd==part->number()) throw runtime_error("W lost. :(");
659  } else if (absId<10 or absId==21) {
660  PartonAppend(PseudoJettify(part),part,(char) 0,-1);
661  PartonDescend(part);
662  } else {
663  cerr << "Unidentified ME particle." << endl;
664  continue;
665  }
666  ++mHardProcCount;
667  }
668  mPtHat = minPt;
669 
670  /* ttbar events: seek lepton+jets (one w to leptons, one to quarks) */
671  if (mMode>=4) {
672  if ((bCount!=2 or qCount!=2 or topLepts.size()!=2)) {
673  AddMessage("Non-semileptonic ttbar",mWeight);
674  return false;
675  }
676  for (auto &tl : topLepts) if (!LeptonAdd(tl,mLeptonFriend)) return false;
677  }
678 
679  if (mMode==3) --mHardProcCount;
680 
681  /* Sanity checks */
682  if ( (mMode<4 and mHardProcCount !=2)
683  or (mMode==2 and mSpecialIndices.size()!=1)
684  or (mMode==3 and mSpecialIndices.size()!=2)
685  or (mMode>=4 and mHardProcCount != 2) )
686  {
687  throw std::logic_error("Unexpected hard process structure");
688  }
689 
690  return true;
691 }

913 double HerwigTree::Sigma2()
914 {
915  double weightedDiffs[4] = {0,0,0,0};
916  double phi = 0, eta = 0, pT2 = 0;
917 
918  for(size_t q = 0; q != mCutJetParts.size(); ++q) {
919  pT2 += pow(mCutJetParts[q].pt(),2);
920  eta += pow(mCutJetParts[q].pt(),2)*mCutJetParts[q].eta();
921  phi += pow(mCutJetParts[q].pt(),2)*mCutJetParts[q].phi();
922  }
923  eta /= pT2; phi /= pT2;
924 
925  for(unsigned int q = 0; q != mCutJetParts.size(); ++q)
926  {
927  double deltaEta = eta-mCutJetParts[q].eta();
928  double deltaPhi = TVector2::Phi_mpi_pi( phi-mCutJetParts[q].phi() );
929  double pT2Tmp = pow(mCutJetParts[q].pt(),2);
930  weightedDiffs[0] += pT2Tmp*deltaEta*deltaEta;
931  weightedDiffs[3] += pT2Tmp*deltaPhi*deltaPhi;
932  weightedDiffs[1] -= pT2Tmp*deltaEta*deltaPhi;
933  }
934  weightedDiffs[2] = weightedDiffs[1];
935 
936  TMatrixDSymEigen me( TMatrixDSym(2,weightedDiffs) );
937  TVectorD eigenvals = me.GetEigenValues();
938 
939  return sqrt(eigenvals[1]/pT2);
940 }
